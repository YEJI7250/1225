<%

    String id=request.getParameter("id");
   
    //2. 유효성 체크
    if(id == null || id.trim().equals("")){
        response.sendRedirect("student2.jsp");
        return;
       
    }
    //비즈니스 메소드 호출
    int result = st.deleteStudent(id.trim());

        msg = "삭제실패!";
       
    }
 
%>
